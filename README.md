## ASSI 1
```
public class Addition {
    public static void main(String[] args) {

        int a = 10;
        int b = 5;
        int sum;

        sum = a + b;

        System.out.println("First number = " + a);
        System.out.println("Second number = " + b);
        System.out.println("Sum = " + sum);
    }
}
```
